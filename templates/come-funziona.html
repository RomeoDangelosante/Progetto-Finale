Logica Python del bot

La strategia principale analizza la struttura dei massimi e minimi:

def analizza_struttura(candele): highs = [c['high'] for c in candele] lows = [c['low'] for c in candele] # higher highs + higher lows = uptrend if highs[-1] > highs[-2] and \ lows[-1] > lows[-2]: return "buy" # lower highs + lower lows = downtrend elif highs[-1] < highs[-2] and \ lows[-1] < lows[-2]: return "sell" return "hold"
